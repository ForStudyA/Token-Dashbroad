### 2. JS computed 属性
在 setup() 函数中，找一个合适位置（比如 agentsTotal 所在的那行 computed 之前或之后），添加:

const sourceBreakdown = computed(() => {
  const sumBySource = summary.value.by_source || [];
  const modelBySource = models.value.by_source || [];
  if (!sumBySource.length && !modelBySource.length) return [];
  const result = [];
  const allSources = new Set();
  sumBySource.forEach(s => allSources.add(s.source));
  modelBySource.forEach(s => allSources.add(s.source));
  allSources.forEach(src => {
    const sumEntry = sumBySource.find(s => s.source === src) || {};
    const modelEntry = modelBySource.find(s => s.source === src) || {};
    result.push({
      source: src,
      requests: sumEntry.requests || modelEntry.total_requests || 0,
      input: sumEntry.input || 0,
      output: sumEntry.output || 0,
      cache_read: sumEntry.cache_read || 0,
      cache_creation: sumEntry.cache_creation || 0,
      cost: sumEntry.cost || 0,
      models: modelEntry.models || []
    });
  });
  result.sort((a, b) => b.requests - a.requests);
  return result;
});

### 3. return 语句
在 setup() 的 return 对象中添加 sourceBreakdown。找到 return{...} 那行，在花括号内添加 sourceBreakdown。
